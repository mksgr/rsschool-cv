var isPalindrome = function(x) {
    let numberArr = (x + "").split('');

    let firstArr = [];
    let secondArr = [];
    for (let i = 0; i < numberArr.length; i++) {
        firstArr.push(numberArr[i]);
        secondArr.push(numberArr[numberArr.length - i - 1]);
    }

    if (firstArr.join('') == secondArr.join('')) {
        return true;
    } else {
        return false;
    }


};
